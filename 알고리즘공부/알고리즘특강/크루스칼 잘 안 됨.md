# 크루스칼 잘 안 됨

```c++
//#define _CRT_SECURE_NO_WARNINGS
#include<iostream>
#include<algorithm>
#include<queue>
#include<cmath>
#include<string.h>
using namespace std;

int n;
long long pos[1001][2]; // 좌표 저장.  0= x /// 1= y
long double arr[1001][1001]; // 그 가중치.
double e;
int u_set[1001];
int check[1001];
long double answer = 0;
priority_queue < pair<long double, pair<int, int>>> pq;

void change_set(int a, int b) {
	for (int i = 0; i < n; i++) {
		if (u_set[i] == b) {
			u_set[i] = a;
		}
	}
}


void solve() {

	while (!pq.empty()) {
		long double cost = -pq.top().first;
		int a = pq.top().second.first;
		int b = pq.top().second.second;
		pq.pop();

		if (u_set[a] == u_set[b])
			continue;

		if (check[b] == 0) {
			if (check[a] == 0)
				check[a] = 1;


			check[b] = 1;
			change_set(min(u_set[a], u_set[b]), max(u_set[a], u_set[b]));
			answer += cost;


		}
		else if (check[b] == 1)
			continue;

	}

	while (1) {
		bool clear = true;
		long double mini = 0;
		bool change = false;
		int itmp = 0;
		int jtmp = 0;
		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				if (u_set[i] != u_set[j]) { // 다른 그룹이라면
					if (mini == 0 && change == false) {
						mini = arr[i][j];
						change = true;
						itmp = i;
						jtmp = j;
					}
					else {
						mini = min(mini, arr[i][j]);
						if (mini == arr[i][j]) {
							itmp = i;
							jtmp = j;
						}
					}
				}
			}
		}
		answer += mini;
		change_set(min(u_set[itmp], u_set[jtmp]), max(u_set[itmp], u_set[jtmp]));
		int utmp = u_set[0];
		for (int i = 0; i < n; i++) {
			if (u_set[i] != utmp) {
				clear = false;
				break;
			}
		}
		if (clear == true) {
			break;
		}

	}


}





int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);
	int t;
	cin >> t;

	for (int tc = 1; tc <= t; tc++) {
		answer = 0;
		memset(check, 0, 1001);

		//for (int i = 0; i < n; i++) {
		//	check[i] = 0;
		//}



		cin >> n;
		for (int i = 0; i < n; i++) {
			cin >> pos[i][0];
			u_set[i] = i;
		}
		for (int i = 0; i < n; i++) {
			cin >> pos[i][1];
		}
		cin >> e;


		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				long double ltmp = sqrt((abs(pos[i][0] - pos[j][0]) * abs(pos[i][0] - pos[j][0])) +
					(abs(pos[i][1] - pos[j][1]) * abs(pos[i][1] - pos[j][1])));
				arr[i][j] = e * (ltmp * ltmp);
			}
		}


		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {

				pq.push({ -arr[i][j],{i,j} });

			}
		}

		solve();

		cout << "#" << tc << " " << (long long)(round(answer)) << endl;

	}



	return 0;
}
```
